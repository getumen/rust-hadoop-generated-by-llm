# Server-Side Encryption (SSE-S3) Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Implement SSE-S3 (AES-256-GCM envelope encryption) so objects stored in the DFS are always encrypted at rest, with the S3 server managing keys transparently.

**Architecture:** A new `SseManager` in `dfs_common` performs envelope encryption — a random 256-bit DEK per object is encrypted with a master KEK loaded from `SSE_MASTER_KEY` env var. The S3 server encrypts body on `PutObject` and decrypts on `GetObject`; ChunkServer and DFS proto are untouched. The encrypted DEK is stored as a field in the existing `.meta` JSON file (`x-amz-sse-encrypted-dek`).

**Tech Stack:** `aes-gcm 0.10.3` (already in dfs_common), `rand 0.10.0`, `base64`, Axum handlers

**Step 3: Implement SseManager**

Create `dfs/common/src/auth/sse.rs`:

```rust
use crate::auth::AuthError;
use aes_gcm::{
    aead::{Aead, KeyInit},
    Aes256Gcm, Nonce,
};
use base64::{engine::general_purpose::STANDARD, Engine as _};
use rand::RngExt;

pub struct SseManager {
    kek: [u8; 32],
}

impl SseManager {
    pub fn new(kek: [u8; 32]) -> Self {
        Self { kek }
    }

    /// Encrypt plaintext. Returns (ciphertext, base64-encoded encrypted DEK).
    pub fn encrypt_object(&self, plaintext: &[u8]) -> Result<(Vec<u8>, String), AuthError> {
        // Generate a random DEK
        let mut dek = [0u8; 32];
        rand::rng().fill(&mut dek);

        // Encrypt plaintext with DEK
        let data_cipher = Aes256Gcm::new_from_slice(&dek)
            .map_err(|e| AuthError::InternalError(format!("Invalid DEK: {}", e)))?;
        let mut data_nonce_bytes = [0u8; 12];
        rand::rng().fill(&mut data_nonce_bytes);
        let data_nonce = Nonce::from_slice(&data_nonce_bytes);
        let mut ciphertext = data_nonce_bytes.to_vec();
        ciphertext.extend(
            data_cipher
                .encrypt(data_nonce, plaintext)
                .map_err(|e| AuthError::InternalError(format!("Encrypt failed: {}", e)))?,
        );

        // Encrypt DEK with KEK
        let kek_cipher = Aes256Gcm::new_from_slice(&self.kek)
            .map_err(|e| AuthError::InternalError(format!("Invalid KEK: {}", e)))?;
        let mut kek_nonce_bytes = [0u8; 12];
        rand::rng().fill(&mut kek_nonce_bytes);
        let kek_nonce = Nonce::from_slice(&kek_nonce_bytes);
        let mut encrypted_dek = kek_nonce_bytes.to_vec();
        encrypted_dek.extend(
            kek_cipher
                .encrypt(kek_nonce, dek.as_ref())
                .map_err(|e| AuthError::InternalError(format!("DEK encrypt failed: {}", e)))?,
        );

        Ok((ciphertext, STANDARD.encode(encrypted_dek)))
    }

    /// Decrypt ciphertext using the base64-encoded encrypted DEK.
    pub fn decrypt_object(&self, ciphertext: &[u8], dek_b64: &str) -> Result<Vec<u8>, AuthError> {
        // Decrypt DEK with KEK
        let encrypted_dek = STANDARD
            .decode(dek_b64)
            .map_err(|e| AuthError::InternalError(format!("Base64 decode failed: {}", e)))?;
        if encrypted_dek.len() < 12 {
            return Err(AuthError::InternalError("Encrypted DEK too short".into()));
        }
        let kek_cipher = Aes256Gcm::new_from_slice(&self.kek)
            .map_err(|e| AuthError::InternalError(format!("Invalid KEK: {}", e)))?;
        let kek_nonce = Nonce::from_slice(&encrypted_dek[..12]);
        let dek = kek_cipher
            .decrypt(kek_nonce, &encrypted_dek[12..])
            .map_err(|e| AuthError::InternalError(format!("DEK decrypt failed: {}", e)))?;

        // Decrypt ciphertext with DEK
        if ciphertext.len() < 12 {
            return Err(AuthError::InternalError("Ciphertext too short".into()));
        }
        let data_cipher = Aes256Gcm::new_from_slice(&dek)
            .map_err(|e| AuthError::InternalError(format!("Invalid DEK: {}", e)))?;
        let data_nonce = Nonce::from_slice(&ciphertext[..12]);
        data_cipher
            .decrypt(data_nonce, &ciphertext[12..])
            .map_err(|e| AuthError::InternalError(format!("Decrypt failed: {}", e)))
    }
}
```

```rust
async fn put_object(
    state: S3AppState,
    bucket: &str,
    key: &str,
    body: Bytes,
    headers: HeaderMap,
) -> Response {
    let dest_path = format!("/{}/{}", bucket, key);
    let content_sha256 = headers
        .get("x-amz-content-sha256")
        .and_then(|v| v.to_str().ok())
        .unwrap_or("unknown");

    let body = if content_sha256 == "STREAMING-AWS4-HMAC-SHA256-PAYLOAD" {
        decode_chunked_payload(body)
    } else {
        body
    };

    let etag = format!("\"{:x}\"", Md5::digest(&body));

    // SSE: encrypt body if SSE_MASTER_KEY is configured
    let sse_requested = headers
        .get("x-amz-server-side-encryption")
        .and_then(|v| v.to_str().ok())
        .map(|v| v.eq_ignore_ascii_case("AES256"))
        .unwrap_or(false);

    let (write_body, encrypted_dek_opt) =
        if let (true, Some(mgr)) = (sse_requested || state.sse_manager.is_some(), &state.sse_manager) {
            match mgr.encrypt_object(&body) {
                Ok((ciphertext, dek_b64)) => (Bytes::from(ciphertext), Some(dek_b64)),
                Err(e) => {
                    tracing::error!("SSE encryption failed: {}", e);
                    return empty_response(StatusCode::INTERNAL_SERVER_ERROR);
                }
            }
        } else {
            (body, None)
        };

    let mut temp_file = NamedTempFile::new().unwrap();
    if let Err(e) = temp_file.write_all(&write_body) {
        tracing::error!("Failed to write temp file: {}", e);
        return empty_response(StatusCode::INTERNAL_SERVER_ERROR);
    }
    let temp_path = temp_file.path().to_owned();

    let write_result = state.client.create_file(&temp_path, &dest_path).await;
    let write_result = if write_result
        .as_ref()
        .is_err_and(|e| e.to_string().contains("already exists"))
    {
        let _ = state.client.delete_file(&dest_path).await;
        state.client.create_file(&temp_path, &dest_path).await
    } else {
        write_result
    };

    match write_result {
        Ok(_) => {
            let mut meta_map = std::collections::HashMap::new();
            meta_map.insert("ETag".to_string(), etag.clone());
            for (k, v) in headers.iter() {
                let k_str = k.as_str();
                if k_str.starts_with("x-amz-meta-") {
                    if let Ok(v_str) = v.to_str() {
                        meta_map.insert(k_str.to_string(), v_str.to_string());
                    }
                }
            }
            if let Some(dek_b64) = &encrypted_dek_opt {
                meta_map.insert("x-amz-sse-encrypted-dek".to_string(), dek_b64.clone());
            }
            let metadata = Metadata { headers: meta_map };
            if let Ok(json) = serde_json::to_string(&metadata) {
                let meta_path = format!("{}.meta", dest_path);
                let mut meta_temp = NamedTempFile::new().unwrap();
                let _ = meta_temp.write_all(json.as_bytes());
                let _ = state
                    .client
                    .create_file(meta_temp.path(), &meta_path)
                    .await;
            }
            let mut builder = Response::builder()
                .status(StatusCode::OK)
                .header(ETAG, etag);
            if encrypted_dek_opt.is_some() {
                builder = builder.header("x-amz-server-side-encryption", "AES256");
            }
            builder.body(Body::empty()).unwrap()
        }
        Err(e) => {
            tracing::error!("PutObject failed: {}", e);
            empty_response(StatusCode::INTERNAL_SERVER_ERROR)
        }
    }
}
```

**Note:** The original `put_object` has duplicated metadata-writing logic for the overwrite case. The new version consolidates it — the overwrite retry is handled by the `write_result` block before the `match`.

```bash
#!/usr/bin/env bash
set -euo pipefail
ENDPOINT="http://localhost:9000"
AWS="aws --endpoint-url $ENDPOINT --region us-east-1 \
     --aws-access-key-id dummy --aws-secret-access-key dummy \
     --no-verify-ssl"

echo "=== SSE-S3 Integration Test ==="

$AWS s3 mb s3://sse-test-bucket 2>/dev/null || true

CONTENT="secret-data-$(date +%s)"
TMPFILE=$(mktemp)
echo "$CONTENT" > "$TMPFILE"

echo "Uploading with SSE-S3..."
$AWS s3 cp "$TMPFILE" s3://sse-test-bucket/sse-obj.txt \
    --sse AES256

echo "Downloading and verifying..."
OUTFILE=$(mktemp)
$AWS s3 cp s3://sse-test-bucket/sse-obj.txt "$OUTFILE"
RESULT=$(cat "$OUTFILE")
if [ "$RESULT" != "$CONTENT" ]; then
    echo "FAIL: content mismatch. Got: $RESULT"
    exit 1
fi
echo "✓ Content matches after SSE round-trip"

echo "Verifying SSE response header..."
HEADERS=$($AWS s3api get-object --bucket sse-test-bucket \
    --key sse-obj.txt /dev/null 2>&1 || true)
# The SSE header is in the get-object response metadata
HEAD_RESP=$($AWS s3api head-object --bucket sse-test-bucket --key sse-obj.txt)
if ! echo "$HEAD_RESP" | grep -q "AES256"; then
    echo "WARN: x-amz-server-side-encryption: AES256 not found in head-object response"
else
    echo "✓ SSE header present in HeadObject response"
fi

$AWS s3 rb s3://sse-test-bucket --force
rm -f "$TMPFILE" "$OUTFILE"
echo "=== SSE TEST PASSED ==="
```
